import Layout from 'lib/components/layout'
import { Link } from '@zeit-ui/react'

export const meta = {
  title: 'ブロック一括解除 for Twitter Lite',
  date: '2020-01-29T23:12:09+09:00',
  description: '',
  image: '',
}

「Twitter ブロック 一括解除」でググると、上位にこの記事が出てきます。

<Link href="https://www.ncaq.net/2017/08/21/" icon color target="_blank">twitterでブロックしたアカウントを全てブロック解除するやりかた - ncaq</Link>

ここに書かれてある通り、デベロッパーツールでブロック解除ボタンをクリックするスクリプトを書くのが早いです。
ただ、記載されているコードが今の Twitter（Twitter Lite と呼ばれる PWA）では使えませんでした。  
  
そこで、2020年1月現在使えるコードを代わりにシェアします。

```js
document.querySelectorAll("div[aria-label=\"ブロック中\"]").forEach(b => b.click());
```

これもサイトの構造が変化したら使えなくなるかもしれません。もし使えなくて困っている方がいましたら、
[Issue](https://github.com/HelloRusk/blog.hellorusk.net/issues) でお知らせください。






export default ({ children }) => <Layout meta={meta}>{children}</Layout>
